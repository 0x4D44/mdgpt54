+function drawClosedShape(context: CanvasRenderingContext2D, points: Array<[number, number]>): void {
+  context.beginPath();
+  points.forEach(([x, y], index) => {
+    if (index === 0) {
+      context.moveTo(x, y);
+    } else {
+      context.lineTo(x, y);
+    }
+  });
+  context.closePath();
+  context.fill();
+  context.stroke();
